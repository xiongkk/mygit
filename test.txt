$folder1 = "D:\FolderA"
$folder2 = "D:\FolderB"

# 获取两个文件夹的所有文件递归列表
$files1 = Get-ChildItem -Path $folder1 -Recurse -File | Select-Object FullName, Length, LastWriteTime, @{Name="RelativePath";Expression={$_.FullName.Substring($folder1.Length)}}
$files2 = Get-ChildItem -Path $folder2 -Recurse -File | Select-Object FullName, Length, LastWriteTime, @{Name="RelativePath";Expression={$_.FullName.Substring($folder2.Length)}}

# 比较对象
$diff = Compare-Object -ReferenceObject $files1 -DifferenceObject $files2 -Property RelativePath, Length, LastWriteTime -PassThru

# 输出结果
$diff | ForEach-Object {
    if ($_.SideIndicator -eq "=>") {
        Write-Host "【新增/修改】在 FolderB 中: $($_.RelativePath)" -ForegroundColor Green
    } elseif ($_.SideIndicator -eq "<=") {
        Write-Host "【删除/独有】在 FolderA 中: $($_.RelativePath)" -ForegroundColor Red
    }
}
